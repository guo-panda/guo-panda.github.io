解决简单的浏览器兼容问题

  • 兼容性

原生js兼容性

基本上好用的DOM操作或者语法 IE低版本浏览器都不兼容，作为前端最苦逼的就是解决各种浏览器的兼容问题，鉴于IE浏览器很严谨，
很强大，在这分享几个常用的处理兼容性的方法

  • 事件对象 event
    兼容性写法 var event =event||window.event
  • 获得执行事件的对象的id
    兼容性写法 var targetId =event.target?event.target.id:event.srcElement.id 返回的是id所以判断的时候需要在写上 targetId.id了
  • 阻止冒泡
    兼容性写法
    if(event&&event.stopPropagation){
    evetn.stopPropagation();
    }else{
    event.cancelBubble
    }
  • 获取用户选择的内容
    兼容写法， var txt;window.getSelection?txt=window.getSelection().toString():txt=document.selection.creatRange().txt;
  • 纯文本操作
    innerHTML更换成 innertext 或者textcontent只能操作纯文本 ，不能操作标签；不兼容.兼容写法：
    function text(tag,txtVal) {
    if(txtVal!==undefined){
    //设置操作
tag.innerText!==undefined?tag.innerText=txtVal:tag.textContent=txtVal;
    }else {
    //获取操作
return tag.innerText!=undefined?tag.innerText:tag.textContent;
    }
    }
  • 获取样式
    兼容写法，
    function style(tag,datas) {
    if(datas!==undefined){
    var style=tag.style;
for(var k in datas){
    if(typeof  datas[k]){
        style[k]=datas[k]();
    }else{
        style[k]=datas[k];
    }
}
    }else {
    //获取 必须有返回值
return tag.currentStyle?tag.currentStyle:getComputedStyle(tag,null);

  • 获取用户选择的内容
    兼容写法， tag.currentStyle?tag.currentStyle:getComputeStyle(tag,null);
  • 封装通过类名获取DOM节点
    function getClass(classname,tag) {
    //第一步先判断 是不是有没有限定的 标签,如果有 就进行判断
    //一旦没有限定的标签就返回 document；如果有就返回他自己
    //var tag=tag||document.body;//用 的是 短路操作
    //用三元运算符也可以
    tag?tag:tag=document.body;
    //第二步判断是不支持类名操作
    if(document.getElementsByClassName){//getElementsByClassName 可以理解为对象的一个方法 如果支持属性就返回一个 函数
    // 如果不支持就返回 undefined 正好和布尔值相应 所以可用来判断
//如果支持
return tag.getElementsByClassName(classname);
    }else {
    //如果不支持
var arr=[];var tempsarr ,temps;
//获取所有tag标签里面的所有标签 然后进行遍历 分割
var tags=tag.getElementsByTagName("*");
for (var i = 0; i < tags.length; i++) {
    temps=tags[i].className;
    //为了优化代码 判断tempsarr 是不是有类名 如果没有则不进行下一步 如果有就分割 遍历 判断是不和我们需要的相等
    if(temps){  //类名是字符串如果是空 则表示是假 就不进行操作；其余为真
        tempsarr=temps.split(" ")//通过split进行吧字符串转换成数组
        //遍历所有的 tempsarr里的所有元素，然后比对
        for (var j = 0; j < tempsarr.length; j++) {
          //判断是不是我们所需的
            if(tempsarr[j]==classname){
                arr[arr.length]=tags[i]//我们需要获取的是有这个类名的标签；而不类名 我们操作也是标签
                break;
            }
        }

        }
    }
    return arr//返回一个数组 
}

}

  • 封装的是鼠标在页面的坐标 page 以及滚动

    var eventTools={
    //检测 参数对象
    getEvent:function (e) {

    return window.event?window.event:event;

    },
    //检测可视区域的坐标
    getClientX:function (e) {

    return this.getEvent(e).clientX;

    },
    getClientY:function (e) {

    return   this.getEvent(e).clientY;

    },
    //检测滚动
    getScrollLeft:function () {

    return window.pageXOffset||document.body.scrollLeft||document.documentElement.scrollLeft||0;

    },
    getScrollTop:function () {

    return window.pageYOffset||document.body.scrollTop||document.documentElement.scrollTop||0;

    },
    //用page去检测鼠标跟随 //page和client都是事件对象的里面的属性
    getPageX:function (e) {

    return this.getEvent(e).pageX?this.getEvent(e).pageX:this.getScrollLeft()+this.getClientX(e);

    },
    getPageY:function (e) {

    return this.getEvent(e).pageY?this.getEvent(e).pageY:this.getScrollTop()+this.getClientY(e);

    }
    }

  • 缓速运动的封装

    function animate(tag,json,fn) {
    clearInterval(tag.timer);
    tag.timer=setInterval(function () {

    var flag=true;
for(var attr in json){
    if(attr=="opacity"){
        var leader=getStyle(tag,attr)*100;
        var target=json[attr]*100;
        var step=(target-leader)/10;
        step=target>leader?Math.ceil(step):Math.floor(step);
        leader+=step;
        tag.style[attr]=leader/100;
    }else if(attr=="zIndex"){
        tag.style[attr]=json[attr];
    }else {
        var leader=parseInt(getStyle(tag,attr));
        var target=json[attr];
        var step=(target-leader)/10;
        step=target>leader?Math.ceil(step):Math.floor(step);
        leader+=step;
        tag.style[attr]=leader+"px";
    }
    if(leader!=target){
        flag=false;
    };
}
if(flag){
    clearInterval(tag.timer);
    if(fn){
        fn();
    }
}

    },20)
    }
